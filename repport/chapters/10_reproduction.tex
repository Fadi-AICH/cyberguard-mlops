\chapter{Guide de reproduction complet}

\section{Cloner le projet}
\begin{lstlisting}
git clone https://github.com/Fadi-AICH/cyberguard-mlops.git
Set-Location -LiteralPath 'C:\Users\fadia\OneDrive\Bureau\Projet ML'
\end{lstlisting}

\section{Créer le venv Python 3.11}
\begin{lstlisting}
py -3.11 -m venv .venv
.\.venv\Scripts\python.exe -m pip install --upgrade pip
.\.venv\Scripts\python.exe -m pip install -r requirements.txt
.\.venv\Scripts\Activate.ps1
$env:PYTHONPATH='src'
\end{lstlisting}

\section{Relancer les artefacts ML}
\begin{lstlisting}
.\.venv\Scripts\python.exe -m cyberguard_ml.pipeline.ingest_ciciot2023 --rows 3000 --page-size 100 --offset 0
.\.venv\Scripts\python.exe -m cyberguard_ml.pipeline.validate_data
.\.venv\Scripts\python.exe -m cyberguard_ml.pipeline.great_expectations_check
.\.venv\Scripts\python.exe -m cyberguard_ml.pipeline.train_model
.\.venv\Scripts\python.exe -m cyberguard_ml.monitoring.drift_report
.\.venv\Scripts\python.exe -m cyberguard_ml.monitoring.soc_report
\end{lstlisting}

\section{Lancer tous les services}
\begin{lstlisting}
docker compose --profile airflow up -d mlflow api prometheus grafana airflow soc-ui
docker compose --profile airflow ps
\end{lstlisting}

\section{Générer du trafic}
\begin{lstlisting}
.\.venv\Scripts\python.exe scripts\replay_ciciot2023_traffic.py --rows 500 --sleep 0.02 --timeout 30 --retries 2
\end{lstlisting}
